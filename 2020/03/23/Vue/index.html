过滤器

过滤器的作用：

过滤器定义的语法：

Vue.filter('过滤器名称', function(data){//要对传过来的数据做处理});

过滤器调用的格式：

{{ name | 过滤器名称}}

过滤器应用：

<!-- 引入Vue.js -->
    <script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>

    <div id="divid">
        <p>{{ msg | msgFormat('44','55') | test}}</p>

    </div>

    <script type="text/javascript">

        Vue.filter('msgFormat', function (msg, arg, arg2) {
            return msg.replace(/22/g, arg + arg2);
        });

        Vue.filter('test',function (msg) {
            return msg + '======';
        });

        var vm = new Vue({
            el: "#divid",
            data: {
               msg: '112233'
            },
            methods: {}
        });
    </script>

Vue-router

  • 安装

    npm install vue-router --save-dev

  • 使用

    • router/index.js(路由配置文件)

    import VueRouter from 'vue-router' //导入路由

import Hello from '../components/Hello.vue' //模块
import login from '../componnents/login.vue' //模块

Vue.use(VueRouter); //显示声明使用VueRouter

export default new VueRouter({
    routes: [
        //配置路由对象，可多个。
        {
            //路由名称(可选)
            name: 'hello'
            //路由路径
            path: '/hello',
            //重定向(可选)
            redirect: '/home',
            //跳转组件
            component: Hello
            //子路由(可选)
            children: [
                {
                    path: '/login',
                    component: 
                }
            ]
        }
    ]
})

    • main.js(主程序入口文件)

    import Vue from 'vue'
import App from './App.vue'
import router from './router' //添加此处：导入路由组件，它自动扫描里面的路由配置

Vue.config.productionTip = false

new Vue({
      el: '#app',
    router, //添加此处：配置路由
    components: { App },
    template: '<App/>'
})
